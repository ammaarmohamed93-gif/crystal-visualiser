async function callHuggingFaceAI(prompt) { const apiKey = prompt("Please enter your Hugging Face token:"); const response = await fetch('https://api.huggingface.co/models/meta-llama/Llama-2-70b-chat-hf', { method: 'POST', headers: { 'Authorization': `Bearer ${apiKey}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ inputs: prompt, parameters: { max_length: 1000, return_full_text: false, } }) }); const data = await response.json(); return data; }
